def gradientDescentMulti(X, y, theta, alpha, iterations):
    m = len(y)
    cost = np.zeros(iterations)
    for i in range(iterations):
        temp = np.dot(X, theta) - y
        temp = np.dot(X.T, temp)
        temp=(alpha/m)*temp
        #print("temp ",temp)
        theta = theta - temp
        #print("theta in ",i,"iter ",theta)
        cost[i] = computeCost(X, y, theta)  
    return theta,cost
